                    <div style={{ margin:"0 18px 12px", borderRadius:7, padding:"8px 12px", background:lightIce, borderLeft:`3px solid ${iceBlue}` }}>
                      <p style={{ fontSize:10, fontWeight:700, color:blueGray, textTransform:"uppercase", letterSpacing:"0.5px", margin:"0 0 2px" }}>Next step</p>
                      <p style={{ fontSize:11, color:gray2, margin:0, lineHeight:1.5, display:"-webkit-box", WebkitLineClamp:1, WebkitBoxOrient:"vertical", overflow:"hidden" }}>{item.next}</p>
                    </div>
                  )}
                  <div style={{ marginTop:"auto", padding:"10px 18px 13px", display:"flex", justifyContent:"space-between", alignItems:"center", borderTop:`1px solid ${lightIce}` }}>
                    <span style={{ fontSize:11, color:blueGray }}>{item.owner}</span>
                    <div style={{ display:"flex", gap:6, alignItems:"center" }}>
                      <span style={{ fontSize:10, fontWeight:600, padding:"3px 9px", borderRadius:6, background:lightIce, color:blueGray, border:`1px solid ${iceBlue}` }}>{item.deadline}</span>
                      <button onClick={e => openEdit(item, e)} style={{ background:"none", border:`1px solid ${iceBlue}`, borderRadius:6, padding:"3px 9px", fontSize:11, color:blueGray, cursor:"pointer" }}>Editar</button>
                    </div>
                  </div>
                </div>
              ))}
            </div>
            {filtered.length === 0 && (
              <div style={{ textAlign:"center", padding:"80px 0", color:blueGray }}>
                <p style={{ fontSize:14 }}>Nenhum item encontrado.</p>
              </div>
            )}
          </div>

          {/* VIEW MODAL */}
          {sel && !editItem && (
            <div onClick={() => setSel(null)} style={{ position:"fixed", inset:0, background:"rgba(0,20,28,.5)", zIndex:50, display:"flex", alignItems:"center", justifyContent:"center", padding:20 }}>
              <div onClick={e => e.stopPropagation()} style={{ background:white, borderRadius:14, width:"100%", maxWidth:540, maxHeight:"88vh", overflowY:"auto", boxShadow:"0 24px 60px rgba(0,30,40,.25)", borderTop:`4px solid ${teal}` }}>
                <div style={{ padding:"22px 26px 18px", borderBottom:`1px solid ${lightIce}` }}>
                  <div style={{ display:"flex", justifyContent:"space-between", alignItems:"flex-start" }}>
                    <div style={{ flex:1, paddingRight:12 }}>
                      <div style={{ display:"flex", gap:8, alignItems:"center", marginBottom:8 }}>
                        <span style={{ fontSize:10, color:teal, fontWeight:700, textTransform:"uppercase", letterSpacing:"0.5px" }}>{sel.cat}</span>
                        <span style={{ fontSize:10, color:gray3, fontFamily:"monospace" }}>#{sel.id}</span>
                        <StatusPill s={sel.status} />
                      </div>
                      <h2 style={{ fontSize:18, fontWeight:700, color:teal, margin:0, lineHeight:1.3 }}>{sel.title}</h2>
                    </div>
                    <button onClick={() => setSel(null)} style={{ background:lightIce, border:"none", color:gray3, width:30, height:30, borderRadius:"50%", cursor:"pointer", fontSize:15, flexShrink:0 }}>✕</button>
                  </div>
                  <div style={{ display:"flex", gap:16, marginTop:14, alignItems:"center" }}>
                    <span style={{ fontSize:12, color:blueGray }}>Responsável: <b>{sel.owner}</b></span>
                    <div style={{ height:14, width:1, background:iceBlue }} />
                    <span style={{ fontSize:11, fontWeight:600, padding:"3px 10px", borderRadius:6, background:lightIce, color:blueGray, border:`1px solid ${iceBlue}` }}>{sel.deadline}</span>
                  </div>
                </div>
                <div style={{ padding:"20px 26px" }}>
                  {[
                    { label:"Situação Atual", text:sel.current, bg:lightIce, border:iceBlue, color:teal },
                    sel.next    ? { label:"Próximos Passos", text:sel.next,     bg:lightIce, border:iceBlue, color:blueGray } : null,
                    sel.comments? { label:"Comentários",     text:sel.comments, bg:lightIce, border:iceBlue, color:blueGray } : null,
                  ].filter(Boolean).map((s,i) => (
                    <div key={i} style={{ marginBottom:14 }}>
                      <p style={{ fontSize:11, fontWeight:700, color:s.color, textTransform:"uppercase", letterSpacing:"0.8px", margin:"0 0 7px" }}>{s.label}</p>
                      <div style={{ background:s.bg, border:`1px solid ${s.border}`, borderRadius:10, padding:"12px 15px", fontSize:13, color:gray1, lineHeight:1.75 }}>{s.text}</div>
                    </div>
                  ))}
                  <div style={{ display:"flex", gap:10, marginTop:20 }}>
                    <button onClick={e => openEdit(sel, e)} style={{ flex:1, padding:"10px", borderRadius:8, fontSize:13, fontWeight:600, background:teal, color:white, border:"none", cursor:"pointer" }}>Editar item</button>
                    <button onClick={() => setConfirmDel(sel.id)} style={{ padding:"10px 18px", borderRadius:8, fontSize:13, fontWeight:600, background:"none", color:"#A02B1A", border:"1.5px solid #F4A99A", cursor:"pointer" }}>Excluir</button>
                  </div>
                </div>
              </div>
            </div>
          )}

          {/* EDIT / CREATE MODAL */}
          {editItem && (
            <div style={{ position:"fixed", inset:0, background:"rgba(0,20,28,.5)", zIndex:60, display:"flex", alignItems:"center", justifyContent:"center", padding:20 }}>
              <div style={{ background:white, borderRadius:14, width:"100%", maxWidth:520, maxHeight:"92vh", overflowY:"auto", boxShadow:"0 24px 60px rgba(0,30,40,.28)", borderTop:`4px solid ${teal}` }}>
                <div style={{ padding:"20px 26px 16px", borderBottom:`1px solid ${lightIce}`, display:"flex", justifyContent:"space-between", alignItems:"center" }}>
                  <h2 style={{ fontSize:17, fontWeight:700, color:teal, margin:0 }}>{editItem.id.startsWith("NEW") ? "Novo item" : "Editar item"}</h2>
                  <button onClick={() => setEditItem(null)} style={{ background:lightIce, border:"none", color:gray3, width:30, height:30, borderRadius:"50%", cursor:"pointer", fontSize:15 }}>✕</button>
                </div>
                <EditForm key={editItem.id} item={editItem} onSave={saveItem} onCancel={() => setEditItem(null)} />
              </div>
            </div>
          )}

          {/* DELETE CONFIRM */}
          {confirmDel && (
            <div onClick={() => setConfirmDel(null)} style={{ position:"fixed", inset:0, background:"rgba(0,20,28,.55)", zIndex:70, display:"flex", alignItems:"center", justifyContent:"center", padding:20 }}>
              <div onClick={e => e.stopPropagation()} style={{ background:white, borderRadius:14, padding:"28px 32px", maxWidth:380, width:"100%", boxShadow:"0 16px 48px rgba(0,30,40,.25)", textAlign:"center" }}>
                <div style={{ fontSize:32, marginBottom:12 }}>⚠️</div>
                <h3 style={{ color:gray1, margin:"0 0 8px", fontSize:16 }}>Excluir item?</h3>
                <p style={{ color:gray2, fontSize:13, margin:"0 0 24px", lineHeight:1.6 }}>Essa ação não poderá ser desfeita.</p>
                <div style={{ display:"flex", gap:10 }}>
                  <button onClick={() => setConfirmDel(null)} style={{ flex:1, padding:"10px", borderRadius:8, fontSize:13, fontWeight:600, background:lightIce, color:gray2, border:`1px solid ${iceBlue}`, cursor:"pointer" }}>Cancelar</button>
                  <button onClick={() => deleteItem(confirmDel)} style={{ flex:1, padding:"10px", borderRadius:8, fontSize:13, fontWeight:600, background:"#A02B1A", color:white, border:"none", cursor:"pointer" }}>Excluir</button>
                </div>
              </div>
            </div>
          )}

        </div>
      );
    }

    const root = ReactDOM.createRoot(document.getElementById("root"));
    root.render(<App />);
  </script>
